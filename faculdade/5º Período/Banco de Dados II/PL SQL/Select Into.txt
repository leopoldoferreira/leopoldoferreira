declare NUMDEP NUMBER;
DEPNUM DEP%ROWTYPE;
begin 
NUMDEP := &NUMDEP;
select * 
into DEPNUM
from DEP
where NUMDEP = DEP.N_DEP;
EXCEPTION
WHEN NO_DATA_FOUND  THEN DBMS_OUTPUT.PUT_LINE('Departamento não existente');
WHEN TOO_MANY_ROWS THEN DBMS_OUTPUT.PUT_LINE('Departamento encontrado');
end;


declare CPRODUTO PRODUTO%ROWTYPE;
PRODUTOCOD produto.codcategoria%TYPE;
PRODUTON PRODUTO.NOMPRODUTO%TYPE;
begin
PRODUTOCOD := &PRODUTOCOD;
select *
into CPRODUTO
from PRODUTO
where PRODUTO.CODCATEGORIA = PRODUTOCOD;
EXCEPTION
WHEN NO_DATA_FOUND  THEN DBMS_OUTPUT.PUT_LINE('Não encontrado o nome do produto');
WHEN TOO_MANY_ROWS THEN DBMS_OUTPUT.PUT_LINE('Nome produto '||PRODUTON||'');
END;
