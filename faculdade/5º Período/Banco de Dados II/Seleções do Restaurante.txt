\\1 Nome da pessoa e sua respectiva Unidade Federativa
 
select PESSOA.NOMPESSOA, UNIDADE_FEDERATIVA.NOME_UF
FROM PESSOA
JOIN CIDADE ON CIDADE.CODCIDADE = PESSOA.CODCIDADE
JOIN UNIDADE_FEDERATIVA ON UNIDADE_FEDERATIVA.SIGLA_UF = CIDADE.SIGLA_UF;

\\2 Nome da pessoa e a situação da sua mesa (Mas como MESASRESERVA não tem dados, aparece soh o nome das colunas).

SELECT PESSOA.NOMPESSOA, MESASRESERVA.IDFSITUACAO
FROM PESSOA
JOIN RESERVA ON RESERVA.CODPESSOA = PESSOA.CODPESSOA
JOIN MESASRESERVA ON MESASRESERVA.NUMRESERVA = RESERVA.NUMRESERVA;

\\3 Nome do produto e o IDFATIVO da tabela ConversaoUnidade( Como as tabelas estão vazias, soh mostra o nome das colunas).

SELECT PRODUTO.NOMPRODUTO , CONVERSAOUNIDADE.IDFATIVO
FROM PRODUTO 
JOIN UNIDADE ON UNIDADE.CODUNIDADE = PRODUTO.CODUNIDADE
JOIN CONVERSAOUNIDADE ON CONVERSAOUNIDADE.CODUNIDADE = UNIDADE.CODUNIDADE;
